inclusion_str = ""
non_required_inclusion_str = ' U '.join(non_required_inclusion_blocks)
exclusion_str = ' ⊄ '.join(exclusion_blocks)  # Creating the exclusion string

# Construct inclusion_str based on whether there are required inclusion blocks
# and non-required inclusion blocks.

    # Joining all required inclusion blocks with ' ∩ ' and grouping non-required inclusions.
    inclusion_str = f"({' ∩ '.join(required_inclusion_blocks)} ∩ ({non_required_inclusion_str}))"
elif required_inclusion_blocks:
    # Joining all required inclusion blocks with ' ∩ ' if there are no non-required inclusions.
    inclusion_str = ' ∩ '.join(required_inclusion_blocks)
elif non_required_inclusion_str:
    inclusion_str = non_required_inclusion_str

# Construct the final output string
if inclusion_str and exclusion_str:
    return f"{inclusion_str} ⊄ {exclusion_str}"  # Utilizing the exclusion string here
elif inclusion_str:
    return f"{inclusion_str}"
elif exclusion_str:
    return f"⊄ {exclusion_str}"  # And here

    return "No logic defined"
